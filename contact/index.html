#!/user/bin/env python 
""" 
contact.py

from __future__ import print_function 


def contact():
    s = "http://www.mikezawitkowski.com"
    s = s[s.index('m'):]
    print(s[:s.index('z'):] + "@" + s)


if __name__ == "__main__":
    contact()
